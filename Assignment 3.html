In[23]:
def clean_sales(file_name):
    df = pd.read_excel(file_name, skiprows=3, skipfooter=4)
    df = df.melt(id_vars='office', var_name='year', value_name='sales')
    df.rename(columns={'office': 'city'}, inplace=True)
    df.dropna(inplace=True)
    return df
